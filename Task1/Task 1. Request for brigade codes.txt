SELECT DISTINCT
    LISTCODE.ID AS "LISTCODE_ID",
    REF_DEVICE.ID AS "DEVICE_ID"
FROM
    LISTCODE
LEFT JOIN
    DEVICE_USAGE ON DEVICE_USAGE."REF_DEVICE_ID" = REF_DEVICE."ID" AND DEVICE_USAGE."DATE_WORK" BETWEEN DATE '2020-06-01' AND DATE '2020-06-30'
WHERE
    LISTCODE."CODE_BRIGAD" NOT IN (
        SELECT
            REF_DEVICE."ID"
        FROM
            DEVICE_USAGE);
